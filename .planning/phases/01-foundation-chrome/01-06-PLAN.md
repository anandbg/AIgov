    ---
    import { getCollection } from 'astro:content';
    import { DENSITY_THRESHOLD } from '~/config/site';

    // Build-time counts. getCollection returns all entries even with underscore prefix; filter underscore-prefixed (fixtures) out for honest counts.
    const allDocs = await getCollection('docs');
    const stageDocs = allDocs.filter((entry) => entry.id.startsWith('stages/') && !entry.id.includes('_'));
    const glossaryEntries = (await getCollection('glossary')).filter((e) => !e.id.startsWith('_'));
    const snapshotEntries = (await getCollection('snapshots')).filter((e) => !e.id.startsWith('_'));

    const stagesCount = stageDocs.length;
    const glossaryCount = glossaryEntries.length;
    const snapshotsCount = snapshotEntries.length;

    function bar(count: number, target: number, width = 12): string {
      const filledChars = Math.min(width, Math.round((count / target) * width));
      return '█'.repeat(filledChars) + '░'.repeat(width - filledChars);
    }
    function pct(count: number, target: number): number {
      return Math.min(100, Math.round((count / target) * 100));
    }
    ---

- Glossary terms:      [ {glossaryCount} of {DENSITY_THRESHOLD.glossaryTerms}+ ] {bar(glossaryCount, DENSITY_THRESHOLD.glossaryTerms)}  ({pct(glossaryCount, DENSITY_THRESHOLD.glossaryTerms)}%)

    ---
    import { HEARTBEAT } from '~/config/site';
    import { readdir, readFile } from 'node:fs/promises';
    import path from 'node:path';
    import { fileURLToPath } from 'node:url';

    // Resolve .heartbeat directory at the repo root (apps/site is two levels deep)
    const repoRoot = path.resolve(fileURLToPath(import.meta.url), '../../../../../');
    const heartbeatDir = path.join(repoRoot, HEARTBEAT.directory);

    type HeartbeatStatus = {
      source: string;
      lastRun: string;
      ok: boolean;
      expectedCadence: string;
    };

    let heartbeats: HeartbeatStatus[] = [];
    try {
      const files = await readdir(heartbeatDir);
      for (const file of files.filter((f) => f.startsWith('last-run-') && f.endsWith('.json'))) {
        try {
          const raw = await readFile(path.join(heartbeatDir, file), 'utf-8');
          const data = JSON.parse(raw) as Partial<HeartbeatStatus>;
          if (data.source && data.lastRun) {
            heartbeats.push({
              source: data.source,
              lastRun: data.lastRun,
              ok: data.ok ?? true,
              expectedCadence: data.expectedCadence ?? 'unknown',
            });
          }
        } catch { /* skip malformed */ }
      }
    } catch {
      /* directory missing — Phase 1 default, no workflows have run yet */
    }
    ---

    <section class="heartbeat-status" aria-label="Pipeline heartbeat status">
      <h3>Pipeline health</h3>
      {heartbeats.length === 0 ? (
        <p class="heartbeat-empty">No heartbeats recorded yet. The keepalive workflow lands in Phase 1 (plan 01-07); per-source heartbeats land in Phase 3 (Tracking Pipeline).</p>
      ) : (
        <ul class="heartbeat-list">
          {heartbeats.map((h) => (
            <li class={h.ok ? 'heartbeat-ok' : 'heartbeat-fail'}>
              {h.ok ? '✓' : '⚠'} <strong>{h.source}</strong> — last run {h.lastRun}, expected {h.expectedCadence}
            </li>
          ))}
        </ul>
      )}
