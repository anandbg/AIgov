    Step 4 — Create `/Users/anand/AIgov/.github/workflows/heartbeat.yml`:
    ```yaml
    name: Heartbeat

    on:
      schedule:
        - cron: '0 12 * * *'  # Daily 12:00 UTC — matches MAINTENANCE.md operational cadence
      workflow_dispatch:

    concurrency:
      group: heartbeat
      cancel-in-progress: false

    permissions:
      contents: read
      issues: write

    jobs:
      check:
        runs-on: ubuntu-latest
        steps:
          - name: Checkout
            uses: actions/checkout@v4
          - name: Check heartbeats
            id: check
            uses: actions/github-script@v7
            with:
              script: |
                const fs = require('fs');
                const path = require('path');

                const HEARTBEAT_DIR = '.heartbeat';
                const STALE_MULTIPLIER = 2;

                // Map cadence string → expected interval in hours
                // 'weekly (Mon 06:00 UTC)' → 168h × 2 = 336h stale threshold
                // 'daily ...' → 24h × 2 = 48h
                // 'bi-weekly ...' → 336h × 2 = 672h
                function cadenceToHours(cadence) {
                  const c = cadence.toLowerCase();
                  if (c.includes('daily')) return 24;
                  if (c.includes('weekly')) return 168;
                  if (c.includes('bi-weekly') || c.includes('biweekly')) return 336;
                  if (c.includes('monthly')) return 720;
                  return null; // unknown → skip staleness check
                }

                if (!fs.existsSync(HEARTBEAT_DIR)) {
                  core.setOutput('stale_count', 0);
                  core.notice('No .heartbeat directory found — nothing to check yet.');
                  return;
                }

                const files = fs.readdirSync(HEARTBEAT_DIR).filter(f => f.startsWith('last-run-') && f.endsWith('.json'));
                const now = Date.now();
                const stale = [];

                for (const file of files) {
                  let data;
                  try {
                    data = JSON.parse(fs.readFileSync(path.join(HEARTBEAT_DIR, file), 'utf-8'));
                  } catch (e) {
                    core.warning(`Malformed JSON in ${file}: ${e.message}`);
                    continue;
                  }
                  const intervalH = cadenceToHours(data.expectedCadence || '');
                  if (intervalH === null) continue;
                  const lastMs = Date.parse(data.lastRun);
                  if (isNaN(lastMs)) continue;
                  const ageH = (now - lastMs) / 1000 / 3600;
                  if (ageH > intervalH * STALE_MULTIPLIER) {
                    stale.push({ source: data.source, lastRun: data.lastRun, ageHours: Math.round(ageH), expectedHours: intervalH });
                  }
                }

                core.setOutput('stale_count', stale.length);
                core.setOutput('stale_json', JSON.stringify(stale));

                if (stale.length > 0) {
                  const body = [
                    `Heartbeat check found ${stale.length} stale source(s):`,
                    '',
                    ...stale.map(s => `- **${s.source}**: last run ${s.lastRun} (${s.ageHours}h ago, expected ≤${s.expectedHours * STALE_MULTIPLIER}h)`),
                    '',
                    'See [MAINTENANCE.md](../blob/main/MAINTENANCE.md) for the operational escalation path.',
                  ].join('\\n');

                  await github.rest.issues.create({
                    owner: context.repo.owner,
                    repo: context.repo.repo,
                    title: `Heartbeat failure — ${stale.length} stale source(s) (${new Date().toISOString().slice(0,10)})`,
                    body,
                    labels: ['heartbeat-failure'],
                  });

                  core.setFailed(`${stale.length} stale heartbeat(s) — issue created.`);
                }
          - name: Notify webhook (optional)
            if: ${{ failure() && env.HEARTBEAT_WEBHOOK_URL != '' }}
            env:
              HEARTBEAT_WEBHOOK_URL: ${{ secrets.HEARTBEAT_WEBHOOK_URL }}
              STALE_JSON: ${{ steps.check.outputs.stale_json }}
            run: |
              curl -fsS -X POST "$HEARTBEAT_WEBHOOK_URL" \
                -H "Content-Type: application/json" \
                -d "{\"text\":\"AIgov heartbeat-failure: $STALE_JSON\"}"
    ```

    The webhook step is gated on (a) the previous step failed (stale count > 0) AND (b) the secret is set — graceful degradation if HEARTBEAT_WEBHOOK_URL is not configured.
