    ```typescript
    import { defineCollection, reference, z } from 'astro:content';
    import { glob } from 'astro/loaders';
    import { docsLoader } from '@astrojs/starlight/loaders';
    import { docsSchema } from '@astrojs/starlight/schema';
    import {
      StageFrontmatter,
      RegulationFrontmatter,
      SnapshotFrontmatter,
      VendorFrontmatter,
      GlossaryFrontmatter,
      StoryFrontmatter,
    } from '@aigov/shared';

    // Starlight ships its own `docs` collection. We extend it for stages.
    const docs = defineCollection({
      loader: docsLoader(),
      schema: docsSchema({
        extend: (context) =>
          // Merge Starlight docs frontmatter with our stage schema for /stages/* pages
          z.object({}).merge(StageFrontmatter.partial().extend({
            // stage pages also accept Starlight's standard fields (title, description) via docsSchema
            regulations: z.array(z.object({
              source: reference('regulations'),
              articles: z.array(z.string()),
              relevance: z.enum(['core', 'supporting', 'related']),
              note: z.string().optional(),
            })).default([]),
          })),
      }),
    });

    const regulations = defineCollection({
      loader: glob({ pattern: '**/index.md', base: './src/content/regulations' }),
      schema: RegulationFrontmatter,
    });

    const snapshots = defineCollection({
      loader: glob({ pattern: '**/snapshots/*.md', base: './src/content/regulations' }),
      schema: SnapshotFrontmatter.extend({
        // Cross-reference back to parent regulation
        source: reference('regulations'),
      }),
    });

    const vendor = defineCollection({
      loader: glob({ pattern: '**/index.md', base: './src/content/vendor' }),
      schema: VendorFrontmatter,
    });

    const glossary = defineCollection({
      loader: glob({ pattern: '**/*.md', base: './src/content/glossary' }),
      schema: GlossaryFrontmatter.extend({
        seeAlso: z.array(z.object({
          kind: z.enum(['stage', 'regulation', 'vendor']),
          slug: z.string(), // resolved at render-time, not via reference() — kind-polymorphic
        })).default([]),
      }),
    });

    const stories = defineCollection({
      loader: glob({ pattern: '**/*.md', base: './src/content/stories' }),
      schema: StoryFrontmatter,
    });

    export const collections = {
      docs, // Starlight's collection (extended with stage schema)
      regulations,
      snapshots,
      vendor,
      glossary,
      stories,
    };
    ```

    Notes on design choices encoded above:
    - The `docs` collection name is required by Starlight 0.39 (it scans this exact name for sidebar/TOC integration). Stages live inside the `docs` collection (under `src/content/docs/stages/`) so that Starlight's sidebar config in `astro.config.mjs` (plan 01-03) automatically renders them as routes. The Stage Zod schema is merged via `docsSchema({ extend: ... })`. This is the Astro 6 + Starlight 0.39 idiomatic pattern.
    - `regulations` and `snapshots` are SEPARATE collections to honor the dated-file-per-snapshot rule from ARCHITECTURE.md (lines 340-389). The snapshot's `source` field is `reference('regulations')` so Astro validates the link at build time.
    - `vendor`, `glossary`, `stories` are flat collections (one file per entity).
    - `seeAlso` in glossary is polymorphic (can point to stage/regulation/vendor) — that polymorphism is incompatible with `reference()` (Astro's reference is single-collection), so it's a string slug validated at render time. Document this exception in the comment.

    Because Starlight owns the `docs` collection, the Phase 2 stage MDX files (CNT-01) will live at `src/content/docs/stages/<slug>.mdx`. For Phase 1, create the directory `src/content/docs/stages/` (and a fixture there in Task 2) so the path exists when CNT-01 lands.

    Run `pnpm --filter @aigov/site check` after writing content.config.ts. If `astro check` complains about the docs collection extension, simplify by removing the `.extend()` step and using docsSchema directly (Phase 2 can refine when real stages land). Document the simplification in the SUMMARY.
